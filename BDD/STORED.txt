CREATE PROCEDURE bajarPaciente
    @paciente int
AS 
	update pacientes
	set fechaBaja=GETDATE()
	where nroPaciente=@paciente;

EXEC bajarPaciente @paciente = 2;

CREATE PROCEDURE modificarDatos
    @paciente int,
    @tel int,
    @estadoCivil varchar(20),
    @cantFamiliares int,
    @plan int,
    @motivo varchar(200)
AS 
BEGIN
	IF @plan = (SELECT planMedico FROM Pacientes WHERE nroPaciente=@paciente) 
	BEGIN
		UPDATE Pacientes
		set 
			telefono=@tel,
			estadoCivil=@estadoCivil,
			familiaresACargo=@cantFamiliares
		where nroPaciente=@paciente;
	END
	ELSE
	BEGIN
		UPDATE pacientes
		set 
			planMedico=@plan,
			telefono=@tel,
			estadoCivil=@estadoCivil,
			familiaresACargo=@cantFamiliares
		where nroPaciente=@paciente;
	
		insert into cambioPlan (idPaciente,idPlan,fecha,motivo)
		values (@paciente,@plan,GETDATE(),@motivo)
	END
END
