create table Profesionales(
	nroProfesional int IDENTITY (1,1) NOT NULL primary key,
	nombre varchar(50),
	apellido varchar(50),
	horaIngreso time,
	horaSalida time,
	practica int references Practicas(Codigo)
)

create table Pacientes(
	nroPaciente int IDENTITY (1,1) NOT NULL primary key,
	nombre varchar(50),
	apellido varchar(50),
	dni int,
	tipoDni varchar(5),
	telefono int,
	mail varchar(50),
	fecNac date,
	sexo char,
	estadoCivil varchar(20),
	familiaresACargo int,
	planMedico int references [plan](id_plan),
	fechaBaja date
)

create table Recetas(
	nroReceta int IDENTITY (1,1) NOT NULL primary key,
	nroPaciente int references Pacientes(nroPaciente) ON DELETE CASCADE,
	nroProfesional int references Profesionales(idProfesional) ON DELETE CASCADE,
	descripcion varchar(200)	
)


create table Turnos(
	nroTurno int IDENTITY (1,1) NOT NULL primary key,
	nombre varchar(50),
	apellido varchar(50),
	nroPaciente int references Pacientes(nroPaciente),
	planMedico int references [plan](id_plan),
	horaLlegada datetime,
	horaAtencion datetime,
	idPrestador int references Prestadores(Id_Prestador),
	importe int,
	observaciones varchar(200)	
)

create table cambioPlan(
	nroCambio int IDENTITY (1,1) NOT NULL primary key,
	nroPaciente int references Pacientes(nroPaciente),
	nroPlan int references [plan](id_plan),
	fecha date,
	motivo varchar(200)
)

create table turnosProf(
	nroTurnoP int IDENTITY (1,1) NOT NULL primary key,
	profesional int references Profesionales(idProfesional),
	fecha date,
	horaDesde time,
	horaHasta time
)
