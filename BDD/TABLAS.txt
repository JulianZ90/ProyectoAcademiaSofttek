create table Pacientes(
	idPaciente int IDENTITY (1,1) NOT NULL primary key,
	nombre varchar(50),
	apellido varchar(50),
	dni int,
	tipoDni varchar(5),
	telefono int,
	mail varchar(50),
	fecNac date,
	sexo char,
	estadoCivil varchar(20),
	familiaresACargo int,
	planMedico int references [plan](id_plan),
	fechaBaja date
)

create table Recetas(
	idReceta int IDENTITY (1,1) NOT NULL primary key,
	idPaciente int references Pacientes(idPaciente) ON DELETE CASCADE,
	idProfesional int references Prestadores(Id_Prestador) ON DELETE CASCADE,
	descripcion varchar(200)	
)

create table Turnos(
	idTurno int IDENTITY (1,1) NOT NULL primary key,
	nombre varchar(50),
	apellido varchar(50),
	idPaciente int references Pacientes(idPaciente),
	planMedico int references [plan](id_plan),
	fechaLlegada date,
	horaLllegada time,
	fechaAtencion date,
	horaAtencion time,
	idPrestador int references Prestadores(Id_Prestador),
	idPractica int references Practicas(Codigo),
	importe int,
	observaciones varchar(200)	
)
create table cambioPlan(
	idCambio int IDENTITY (1,1) NOT NULL primary key,
	idPaciente int references Pacientes(idPaciente),
	idPlan int references [plan](id_plan),
	fecha date,
	motivo varchar(200)
)

create table turnosPrest(
	isTurnoP int IDENTITY (1,1) NOT NULL primary key,
	profesional int references Profesionales(Id_Prestador),
	fecha date,
	horaDesde time,
	horaHasta time
)
