create table Profesionales(
	idProfesional int IDENTITY (1,1) NOT NULL primary key,
	nombre varchar(50),
	apellido varchar(50),
	horaIngreso time,
	horaSalida time,
	practica int references Practicas(idPractica)
)

create table Pacientes(
	nroPaciente int IDENTITY (1,1) NOT NULL primary key,
	nombre varchar(50),
	apellido varchar(50),
	dni int,
	tipoDni varchar(5),
	telefono int,
	mail varchar(50)
	fecNac date,
	sexo char,
	estadoCivil varchar(20),
	cantHijos int,
	planMedico int references Planes(idPlan)
)

create table Recetas(
	idReceta int IDENTITY (1,1) NOT NULL primary key,
	idPaciente int references Pacientes(idPaciente) ON DELETE CASCADE,
	idProfesional int references Profesionales(idProfesional) ON DELETE CASCADE,
	descripcion varchar(200)	
)

create table Turnos(
	nroTurno int IDENTITY (1,1) NOT NULL primary key,
	nombre varchar(50),
	apellido varchar(50),
	nroAfiliado int references Pacientes(nroPaciente),
	planMedico int references Planes(idPlan),
	fechaLlegada datetime,
	fechaAtencion datetime,
	idPrestador int references Prestadores(idPrestador),
	importe int,
	observaciones varchar(200)	
)

create table cambioPlan(
	idCambio int IDENTITY (1,1) NOT NULL primary key,
	idPaciente int references Pacientes(idPaciente),
	idPlan int references Planes(idPlan),
	fecha date,
	motivo varchar(200)
)

create table turnosProf(
	idTurnoP int IDENTITY (1,1) NOT NULL primary key,
	profesional int references Profesionales(idProfesional),
	fecha date,
	horaDesde time,
	horaHasta time
)

create table exPacientes(
	idEPaciente int NOT NULL primary key,
	nombre varchar(50),
	apellido varchar(50),
	dni int,
	tipoDni varchar(5),
	telefono int,
	mail varchar(50)
	fecNac date,
	sexo char,
	estadoCivil varchar(20),
	cantHijos int,
	planMedico int references Planes(idPlan)
	fechaBaja date;
)
